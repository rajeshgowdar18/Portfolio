Project 1: Dijistra Algorithm


                #include <stdio.h>
                #define inf 9999
                int visit[100], count, mindistance, nextnode, i, j,distance[100],n,u,a[100][100];
                void Dijkstra() {
                  for (i = 0; i < n; i++) {
                    distance[i] = a[u][i];
                    visit[i] = 0;
                  }
                  distance[u] = 0;
                  visit[u] = 1;
                  count = 1;
                  while (count < n - 1) {
                    mindistance = inf;
                    for (i = 0; i < n; i++)
                      if (distance[i] < mindistance && !visit[i]) 
                      {
                        mindistance = distance[i];
                        nextnode = i;
                      }
                    visit[nextnode] = 1;
                    for (i = 0; i < n; i++)
                      if (!visit[i])
                        if (mindistance + a[nextnode][i] < distance[i]) {
                          distance[i] = mindistance + a[nextnode][i];
                        }
                    count++;
                  }
                  for (i = 0; i < n; i++)
                    if (i != u) {
                      printf("\nDistance from source to %d: %d", i+1, distance[i]);
                    }
                }
                void main() {
                  printf("Enter the number of vertices:");
                  scanf("%d",&n);
                  printf("Enter the matrix:\n");
                  for(i=0;i<n;i++)
                    {
                        for(j=0;j<n;j++)
                        {
                        scanf("%d",&a[i][j]);
                        if (a[i][j] == 0)
                         a[i][j] = inf;
                        }
                    }
                printf("enter the source:");
                scanf("%d",&u);
                --u;
                Dijkstra();
                }

Project 2: Nqueen program


                #include<stdio.h>
                #include<stdlib.h>
                int a[30],count=0,n,i,j,k=1;
                int place(int pos)
                {
                for(i=1;i<pos;i++)
                {
                    if((a[i]==a[pos])||((abs(a[i]-a[pos])==abs(i-pos))))
                    return 0;
                }
                return 1;
                }
                
                void print_sol()
                {
                count++;
                printf("\n\nSolution #%d:\n",count);
                for(i=1;i<=n;i++)
                {
                for(j=1;j<=n;j++)
                {
                if(a[i]==j)
                printf("Q\t");
                else
                printf("*\t");
                }
                printf("\n");
                }
                }
                
                
                void queen()
                {
                a[k]=0;
                while(k!=0)
                {
                a[k]=a[k]+1;
                while((a[k]<=n)&&!place(k))
                a[k]++;
                if(a[k]<=n)
                {
                if(k==n)
                print_sol();
                else
                {
                k++;
                a[k]=0;
                }
                }
                else
                k--;
                }
                }
                
                void main()
                {
                printf("Enter the number of Queens\n");
                scanf("%d",&n);
                queen();
                printf("\nTotal solutions=%d",count);
                }
